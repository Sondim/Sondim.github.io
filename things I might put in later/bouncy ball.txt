<script>
    function updateBouncePhysics() {
        const v = document.getElementById('bounce-slider').value;
        const b = document.getElementById('interactive-ball');
        b.style.setProperty('--squash-ratio', v);
        b.style.setProperty('--stretch-ratio', 1 + ((v - 1) * 0.5));
        b.style.setProperty('--bounce-height', '-120px');
    }

    function confirmBounce() {
        const v = document.getElementById('bounce-slider').value;
        const k = v <= 1.25 ? 'rigid' : v >= 1.75 ? 'loose' : 'bouncy';
        triggerRes(resultsData[k], "Physics Audit Complete");
    }

    // Initializer that was at the bottom of your script
    updateBouncePhysics();
</script>
